int findDuplicate(vector<int>& nums){
int low = 1,high = nums.size()-1;
int mid,cnt;
while(low <= high){
  cnt = 0;
  mid = (low + high) /2;
  for(auto n: nums){
    if(n <= mid) ++cnt;
  }
  if(cnt <= mid) low = mid +1;
  else high = mid - 1;
}
return low;
}

int findDuplicate(vector<int>& nums)
{
  int t = nums[0];
  int h = nums[nums[0]];
  while (t != h){
    t = nums[t];
    h = nums[nums[h]];
  }

  h = 0;
  while (t != h){
    t = nums[t];
    h = nums[h];
  }
  return t;
}
